    <script>

        let submitted = false;

 

        // Cập nhật danh sách 12 cuốn sách theo yêu cầu của bạn (9 cuốn yêu cầu + 3 cuốn bổ sung cùng chủ đề)

        const booksData = [

            // 3 cuốn bổ sung để tròn 12 cuốn (phân làm 2 trang, mỗi trang 6 cuốn)

            { id: 10, title: "Wuthering Heights", author: "Emily Brontë", price: 220000, img: "https://images-na.ssl-images-amazon.com/images/I/81A-f9pS69L._AC_UL600_SR600,400_.jpg" },

            { id: 11, title: "Pride and Prejudice", author: "Jane Austen", price: 190000, img: "https://images-na.ssl-images-amazon.com/images/I/81wKovwY6ML._AC_UL600_SR600,400_.jpg" },

            { id: 12, title: "Great Expectations", author: "Charles Dickens", price: 250000, img: "https://images-na.ssl-images-amazon.com/images/I/619864YkSlL._AC_UL600_SR600,400_.jpg" }

        ];

 

        let filteredBooks = [...booksData];

        let currentPage = 1;

        const itemsPerPage = 6; // Phân trang 6 loại sản phẩm một trang cố định

        let cart = {};

 

        function renderBooks() {

            const grid = document.getElementById('booksGrid');

            grid.innerHTML = "";

           

            const startIndex = (currentPage - 1) * itemsPerPage;

            const endIndex = startIndex + itemsPerPage;

            const pageItems = filteredBooks.slice(startIndex, endIndex);

 

            if (pageItems.length === 0) {

                grid.innerHTML = `<div class="col-span-full text-center py-12 text-gray-500">Không tìm thấy cuốn sách kinh điển nào khớp từ khóa.</div>`;

                return;

            }

 

            pageItems.forEach(book => {

                const bookCard = `

                    <div class="bg-white rounded-xl shadow-sm overflow-hidden hover:shadow-lg transition duration-300 flex flex-col border border-stone-200/60">

                        <div class="h-72 bg-stone-100 overflow-hidden flex items-center justify-center p-4">

                            <img src="${book.img}" alt="${book.title}" class="h-full object-contain hover:scale-105 transition duration-300 shadow-md">

                        </div>

                        <div class="p-5 flex-grow flex flex-col justify-between bg-stone-50/50">

                            <div>

                                <h3 class="text-base font-serif font-bold text-stone-950 line-clamp-1">${book.title}</h3>

                                <p class="text-xs text-stone-500 mb-2">By ${book.author}</p>

                            </div>

                            <div class="mt-4 flex items-center justify-between">

                                <span class="text-amber-900 font-extrabold text-base">${book.price.toLocaleString('vi-VN')} đ</span>

                                <button onclick="addToCart(${book.id})" class="bg-amber-950 hover:bg-amber-800 text-white text-xs font-bold px-3 py-2 rounded shadow transition duration-200">

                                    <i class="fa-solid fa-plus mr-1"></i> Thêm vào giỏ

                                </button>

                            </div>

                        </div>

                    </div>

                `;

                grid.innerHTML += bookCard;

            });

            renderPagination();

        }

 

        function renderPagination() {

            const controls = document.getElementById('paginationControls');

            controls.innerHTML = "";

            const totalPages = Math.ceil(filteredBooks.length / itemsPerPage);

 

            if (totalPages <= 1) return;

 

            for (let i = 1; i <= totalPages; i++) {

                const activeClass = (i === currentPage) ? 'bg-amber-950 text-white' : 'bg-white text-stone-700 hover:bg-stone-100';

                controls.innerHTML += `<button onclick="goToPage(${i})" class="px-4 py-2 border border-stone-300 rounded font-medium transition ${activeClass}">${i}</button>`;

            }

        }

 

        function goToPage(page) {

            currentPage = page;

            renderBooks();

            window.scrollTo({ top: 0, behavior: 'smooth' });

        }

 

        function handleSearch() {

            const query = document.getElementById('searchInput').value.toLowerCase().trim();

            filteredBooks = booksData.filter(book =>

                book.title.toLowerCase().includes(query) ||

                book.author.toLowerCase().includes(query)

            );

            currentPage = 1;

            renderBooks();

        }

 

        function addToCart(id) {

            cart[id] = (cart[id] || 0) + 1;

            updateCartStatus();

            const cartIcon = document.getElementById('cartCount');

            cartIcon.classList.add('scale-125', 'bg-red-600');

            setTimeout(() => cartIcon.classList.remove('scale-125', 'bg-red-600'), 300);

        }

 

        function updateCartStatus() {

            let totalCount = 0;

            let totalPrice = 0;

            let detailsString = "";

 

            const listContainer = document.getElementById('cartItemsList');

            listContainer.innerHTML = "";

 

            Object.keys(cart).forEach(id => {

                const book = booksData.find(b => b.id == id);

                const quantity = cart[id];

                totalCount += quantity;

                totalPrice += book.price * quantity;

                detailsString += `${book.title} (x${quantity}) | `;

 

                listContainer.innerHTML += `

                    <div class="flex justify-between items-center py-2">

                        <div class="max-w-[70%]">

                            <p class="font-serif font-semibold text-sm text-stone-900 truncate">${book.title}</p>

                            <p class="text-xs text-stone-500">${book.price.toLocaleString('vi-VN')}đ x ${quantity}</p>

                        </div>

                        <div class="flex items-center gap-2">

                            <button onclick="changeQuantity(${id}, -1)" class="text-stone-400 hover:text-red-500 px-1"><i class="fa-solid fa-minus text-xs"></i></button>

                            <span class="text-sm font-bold w-4 text-center">${quantity}</span>

                            <button onclick="changeQuantity(${id}, 1)" class="text-stone-400 hover:text-emerald-600 px-1"><i class="fa-solid fa-plus text-xs"></i></button>

                        </div>

                    </div>

                `;

            });

 

            document.getElementById('cartCount').innerText = totalCount;

            document.getElementById('cartTotal').innerText = `Tổng cộng: ${totalPrice.toLocaleString('vi-VN')} đ`;

            document.getElementById('hiddenOrderDetails').value = detailsString;

        }

 

        function changeQuantity(id, delta) {

            cart[id] += delta;

            if (cart[id] <= 0) delete cart[id];

            updateCartStatus();

        }

 

        function toggleCartModal() {

            const modal = document.getElementById('cartModal');

            modal.classList.toggle('opacity-0');

            modal.classList.toggle('pointer-events-none');

            document.body.classList.toggle('modal-active');

        }

 

        function showPaymentStep() {

            toggleCartModal();

            const payModal = document.getElementById('paymentModal');

            payModal.classList.remove('opacity-0', 'pointer-events-none');

            cart = {};

            updateCartStatus();

            document.getElementById('orderForm').reset();

            submitted = false;

        }

 

        function closeAllModals() {

            const payModal = document.getElementById('paymentModal');

            payModal.classList.add('opacity-0', 'pointer-events-none');

            document.body.classList.remove('modal-active');

        }

 

        window.onload = function() {

            renderBooks();

        };
